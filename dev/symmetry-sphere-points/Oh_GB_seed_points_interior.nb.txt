nWedgePoints[q_Integer?Positive] := (q - 1) (q - 2)/2;

(* Generates points in the irreducible wedge of the Oh point group and\
 returns their {(x, y, z}, {theta, phi}} coordinatees.  \
You have two options: \
1) to include the points interior to the wedge only, \
or 2) to include points along the mirror planes  *)

Clear[ohWedgePoints];

ohWedgePoints[q_Integer?Positive] := Module[
  {A, B, C, pts, p, pn, theta, phi},
  
  A = {1, 0, 0};
  B = {1, 1, 0}/Sqrt[2];
  C = {1, 1, 1}/Sqrt[3];
  
  pts = Flatten[
    Table[
     
     p = (i/q) A + (j/q) B + (1 - i/q - j/q) C;
     pn = Normalize[p] (* cartessian coordinates on the unit cell *);
     
     phi = ArcTan[pn[[1]], pn[[2]]];
     theta = ArcCos[pn[[3]]];
     
     {
      pn,
      {phi, theta}
      },
     
     (* points interior to the wedge *)
     {i, 1, q - 2},
     {j, 1, q - i - 1}
     (*  for the points along the mirror planes *)
     (* {i,0,q},
     {j,0,q-i}*)
     ],
    1
    ]
  ]



pts = ohWedgePoints[16];

plotOhWedgeOnSphere[q_Integer?Positive] := Module[
  {pts, cartPts, wedgeEdges},
  
  pts = ohWedgePoints[q];
  cartPts = pts[[All, 1]];
  
  wedgeEdges = {
    Line[Table[{Cos[t], Sin[t], 0}, {t, 0, Pi/4, Pi/200}]],
    
    Line[Table[
      Normalize[(1 - t) {1, 0, 0} + t {1, 1, 0}/Sqrt[2]],
      {t, 0, 1, 1/100}
      ]],
    
    Line[Table[
      Normalize[(1 - t) {1, 1, 0}/Sqrt[2] + t {1, 1, 1}/Sqrt[3]],
      {t, 0, 1, 1/100}
      ]],
    
    Line[Table[
      Normalize[(1 - t) {1, 0, 0} + t {1, 1, 1}/Sqrt[3]],
      {t, 0, 1, 1/100}
      ]]
    };
  
  Graphics3D[
   {
    Opacity[0.12], Sphere[{0, 0, 0}, 1],
    
    Black, Thick, wedgeEdges,
    
    Red, PointSize[0.018], Point[cartPts]
    },
   Boxed -> False,
   Axes -> True,
   AxesLabel -> {"x", "y", "z"},
   PlotRange -> 1.05,
   ImageSize -> Large
   ]
  ]

plotOhWedgeOnSphere[16]

(* generates the full sphere symmetry appropriate gradient path seeds o\
n a unit sphere *)

Clear[ohImages];

ohImages[p_] := DeleteDuplicates[
   Flatten[
    Table[
     sign*p[[perm]],
     {perm, Permutations[{1, 2, 3}]}, {sign, Tuples[{-1, 1}, 3]}
     ],
    1
    ]
   ];


fullSpherePts = DeleteDuplicates[Flatten[ohImages /@ cartPts, 1]];



Graphics3D[
 {
  Opacity[0.1], Sphere[],
  Red, PointSize[0.01],
  Point[fullSpherePts]
  },
 Boxed -> False,
 Axes -> True,
 ImageSize -> Large
 ]
